// ============================================================ // Telegram.gs — Bot API Communication Layer // All outgoing messages to Telegram go through here // ============================================================ var TELEGRAM_API_BASE = "https://api.telegram.org/bot"; /** * Escapes Telegram legacy Markdown (parse_mode: "Markdown") reserved * characters in user-controlled text before interpolating it into a * message string. */ function escapeMarkdown(text) { if (text === null || text === undefined) return ""; return String(text).replace(/([_*`\[\]])/g, "\\$1"); } /** * Send a Markdown message with auto-escaped interpolations. * Usage: sendMarkdownMessage(chatId, "Hello *{name}*", { name: userName }); * All {param} values are escaped for Markdown. */ function sendMarkdownMessage(chatId, template, params) { var text = template.replace(/\{(\w+)\}/g, function(match, key) { var val = params[key]; return val !== undefined ? escapeMarkdown(String(val)) : ""; }); return sendTelegramMessage(chatId, text); } /** * Send Markdown with inline keyboard, auto-escaping template params. */ function sendMarkdownMessageWithInlineKeyboard(chatId, template, params, buttons) { var text = template.replace(/\{(\w+)\}/g, function(match, key) { var val = params[key]; return val !== undefined ? escapeMarkdown(String(val)) : ""; }); return sendTelegramMessageWithInlineKeyboard(chatId, text, buttons); } /** * Send a simple text message (Markdown formatted) * Prefer sendMarkdownMessage for user-controlled content. */ function sendTelegramMessage(chatId, text) { try { var url = TELEGRAM_API_BASE + CONFIG.BOT_TOKEN + "/sendMessage"; var payload = { chat_id: String(chatId), text: text, parse_mode: "Markdown", }; var options = { method: "post", contentType: "application/json", payload: JSON.stringify(payload), muteHttpExceptions: true, }; var response = UrlFetchApp.fetch(url, options); var result = JSON.parse(response.getContentText()); if (!result.ok) { Logger.log("Telegram sendMessage failed: " + JSON.stringify(result)); } return result; } catch (err) { Logger.log("sendTelegramMessage error: " + err.message); return null; } } /** * Send message with inline keyboard buttons * buttons = array of rows, each row = array of button objects * button object: { text: "Button Label", callback_data: "some_data" } */ function sendTelegramMessageWithInlineKeyboard(chatId, text, buttons) { try { var url = TELEGRAM_API_BASE + CONFIG.BOT_TOKEN + "/sendMessage"; var payload = { chat_id: String(chatId), text: text, parse_mode: "Markdown", reply_markup: { inline_keyboard: buttons, }, }; var options = { method: "post", contentType: "application/json", payload: JSON.stringify(payload), muteHttpExceptions: true, }; var response = UrlFetchApp.fetch(url, options); var result = JSON.parse(response.getContentText()); if (!result.ok) { Logger.log("Telegram inline keyboard failed: " + JSON.stringify(result)); } return result; } catch (err) { Logger.log("sendTelegramMessageWithInlineKeyboard error: " + err.message); return null; } } /** * Answer a callback query (required to dismiss loading spinner on inline button) */ function answerCallbackQuery(callbackQueryId, text) { try { var url = TELEGRAM_API_BASE + CONFIG.BOT_TOKEN + "/answerCallbackQuery"; var payload = { callback_query_id: callbackQueryId, text: text || "", show_alert: false, }; var options = { method: "post", contentType: "application/json", payload: JSON.stringify(payload), muteHttpExceptions: true, }; UrlFetchApp.fetch(url, options); } catch (err) { Logger.log("answerCallbackQuery error: " + err.message); } } /** * Edit an existing message text (used to update after owner approves/rejects) */ function editTelegramMessage(chatId, messageId, newText) { try { var url = TELEGRAM_API_BASE + CONFIG.BOT_TOKEN + "/editMessageText"; var payload = { chat_id: String(chatId), message_id: messageId, text: newText, parse_mode: "Markdown", }; var options = { method: "post", contentType: "application/json", payload: JSON.stringify(payload), muteHttpExceptions: true, }; UrlFetchApp.fetch(url, options); } catch (err) { Logger.log("editTelegramMessage error: " + err.message); } } /** * Register this GAS Web App URL as Telegram webhook * Run this function ONCE from GAS editor after deploying */ function registerWebhook() { var webAppUrl = ScriptApp.getService().getUrl(); if (webAppUrl.indexOf("/dev") !== -1) { Logger.log("❌ REFUSING to register a /dev URL as webhook: " + webAppUrl); Logger.log("This usually happens when registerWebhook() is run directly from the editor without an active Web App deployment in scope."); Logger.log("Fix: Deploy → Manage deployments → confirm a Web App deployment exists and is the one currently selected/active, then re-run this function."); Logger.log("If this keeps happening, hardcode your real /exec URL below instead of relying on ScriptApp.getService().getUrl() — see PRODUCTION_WEBHOOK_URL."); return { ok: false, error: "REFUSED_DEV_URL", url: webAppUrl }; } Logger.log("Registering webhook with URL: " + webAppUrl); var url = TELEGRAM_API_BASE + CONFIG.BOT_TOKEN + "/setWebhook"; var payload = { url: webAppUrl, allowed_updates: ["message", "callback_query"], drop_pending_updates: true, }; var options = { method: "post", contentType: "application/json", payload: JSON.stringify(payload), muteHttpExceptions: true, }; var response = UrlFetchApp.fetch(url, options); var result = JSON.parse(response.getContentText()); Logger.log("Webhook registration: " + JSON.stringify(result)); return result; } /** * Use this INSTEAD of registerWebhook() if ScriptApp.getService().getUrl() * keeps returning a /dev URL for you. Paste your real, confirmed /exec * Web App deployment URL below (the SAME one you put in index.html's * GAS_URL), then run this function. */ function registerWebhookManualUrl() { var webAppUrl = "https://script.google.com/macros/s/AKfycbxl2oilhA7Wgy4MyqgC4QOE_x9QhkcaEG4s7_N2AbHvFRjfiylEI-MuxYDKwksAb-2QBQ/exec"; if (webAppUrl.indexOf("PASTE_YOUR_EXACT_EXEC_URL_HERE") !== -1) { Logger.log("❌ Edit this function first: replace the placeholder with your real /exec URL."); return { ok: false, error: "PLACEHOLDER_NOT_REPLACED" }; } if (webAppUrl.indexOf("/exec") === -1) { Logger.log("❌ This URL doesn't end in /exec — double-check you copied the Web App deployment URL, not a /dev URL."); return { ok: false, error: "NOT_AN_EXEC_URL" }; } var url = TELEGRAM_API_BASE + CONFIG.BOT_TOKEN + "/setWebhook"; var payload = { url: webAppUrl, allowed_updates: ["message", "callback_query"], drop_pending_updates: true, }; var options = { method: "post", contentType: "application/json", payload: JSON.stringify(payload), muteHttpExceptions: true, }; var response = UrlFetchApp.fetch(url, options); var result = JSON.parse(response.getContentText()); Logger.log("Webhook registration (manual URL): " + JSON.stringify(result)); return result; } /** * Delete webhook (useful for testing locally) */ function deleteWebhook() { var url = TELEGRAM_API_BASE + CONFIG.BOT_TOKEN + "/deleteWebhook"; var options = { method: "post", muteHttpExceptions: true }; var response = UrlFetchApp.fetch(url, options); Logger.log("Delete webhook: " + response.getContentText()); } /** * Handle bot update (from doPost in Code.gs) * Processes /start, /delete commands and inline button callbacks */ function handleBotUpdate(update) { try { if (update && update.update_id) { var cache = CacheService.getScriptCache(); var botPrefix = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, CONFIG.BOT_TOKEN).slice(0, 8).join(""); var seenKey = "upd_" + botPrefix + "_" + String(update.update_id); if (cache.get(seenKey)) { return ContentService.createTextOutput("OK"); } cache.put(seenKey, "1", 180); } if (update.callback_query) { handleCallbackQuery(update.callback_query); return ContentService.createTextOutput("OK"); } if (update.message) { var msg = update.message; var chatId = msg.chat.id; var userId = msg.from.id; var text = (msg.text || "").trim(); if (text.startsWith("/start")) { var parts = text.split(" "); if (parts.length > 1) { var startParam = parts[1]; if (startParam.startsWith("INV_")) { var inviteToken = startParam.substring(4); var miniAppUrl = "https://t.me/" + getBotUsername() + "/app?startapp=INV_" + inviteToken; sendTelegramMessage(chatId, "👋 You've been invited to join a hotel!\n\nTap below to complete setup:", ); sendMiniAppButton(chatId, "Open MyNirvaha", miniAppUrl); } else { sendWelcomeMessage(chatId, msg.from); } } else { sendWelcomeMessage(chatId, msg.from); } } else if (text === "/help") { sendTelegramMessage(chatId, "📋 *MyNirvaha Help*\n\n" + "• /start — Open the app\n" + "• Use the Mini App for all operations\n\n" + "Need support? Contact your admin." ); } } } catch (err) { Logger.log("handleBotUpdate error: " + err.message); } return ContentService.createTextOutput("OK"); } /** * Handle inline keyboard button presses */ function handleCallbackQuery(callbackQuery) { var data = callbackQuery.data || ""; var userId = callbackQuery.from.id; var chatId = callbackQuery.message.chat.id; var messageId = callbackQuery.message.message_id; var callbackId = callbackQuery.id; var cache = CacheService.getScriptCache(); var seenKey = "cbq_" + callbackId; if (cache.get(seenKey)) { answerCallbackQuery(callbackId, "Already processed"); return; } cache.put(seenKey, "1", 120); try { // ── High-value entry: APPROVE ── if (data.startsWith("APPROVE_")) { var logId = data.substring(8); var logResult = getLogById(logId); if (!logResult) { answerCallbackQuery(callbackId, "⚠️ Entry not found"); return; } if (!isOwnerOfHotel(userId, logResult.data.hotelId)) { answerCallbackQuery(callbackId, "❌ You are not authorized"); return; } var hotelId = logResult.data.hotelId; var lockAcquired = acquireHotelLock(hotelId); if (!lockAcquired) { answerCallbackQuery(callbackId, "⏳ Busy processing another request for this hotel — please tap again in a moment."); return; } try { var freshLogResult = getLogById(logId); if (!freshLogResult) { answerCallbackQuery(callbackId, "⚠️ Entry not found"); return; } if (freshLogResult.data.status === "DELETED") { answerCallbackQuery(callbackId, "⚠️ This entry was already rejected — cannot approve."); editTelegramMessage(chatId, messageId, callbackQuery.message.text + "\n\n❌ *Already rejected*" ); return; } if (freshLogResult.data.status === "PENDING_DELETE") { answerCallbackQuery(callbackId, "⚠️ This entry has a pending delete request — use the delete buttons instead."); return; } if (freshLogResult.data.status === "ACTIVE") { answerCallbackQuery(callbackId, "✅ Entry already approved"); editTelegramMessage(chatId, messageId, callbackQuery.message.text + "\n\n✅ *Already approved*" ); return; } if (freshLogResult.data.status !== "ACTIVE") { updateLogStatus(freshLogResult.rowIndex, "ACTIVE"); } answerCallbackQuery(callbackId, "✅ Entry approved"); editTelegramMessage(chatId, messageId, callbackQuery.message.text + "\n\n✅ *Approved by owner*" ); } finally { releaseHotelLock(hotelId); } } // ── High-value entry: REJECT ── else if (data.startsWith("REJECT_")) { var logId = data.substring(7); var logResult = getLogById(logId); if (!logResult) { answerCallbackQuery(callbackId, "⚠️ Entry not found"); return; } if (!isOwnerOfHotel(userId, logResult.data.hotelId)) { answerCallbackQuery(callbackId, "❌ You are not authorized"); return; } var hotelId = logResult.data.hotelId; var lockAcquired = acquireHotelLock(hotelId); if (!lockAcquired) { answerCallbackQuery(callbackId, "⏳ Busy processing another request for this hotel — please tap again in a moment."); return; } try { var freshLogResult = getLogById(logId); if (!freshLogResult) { answerCallbackQuery(callbackId, "⚠️ Entry not found"); return; } if (freshLogResult.data.status === "DELETED") { answerCallbackQuery(callbackId, "⚠️ This entry was already rejected — no action taken."); editTelegramMessage(chatId, messageId, callbackQuery.message.text + "\n\n❌ *Already rejected*" ); return; } if (freshLogResult.data.status === "PENDING_DELETE") { answerCallbackQuery(callbackId, "⚠️ This entry has a pending delete request — use the delete buttons instead."); return; } if (freshLogResult.data.status === "ACTIVE") { answerCallbackQuery(callbackId, "✅ Entry already approved"); editTelegramMessage(chatId, messageId, callbackQuery.message.text + "\n\n✅ *Already approved*" ); return; } updateLogStatus(freshLogResult.rowIndex, "DELETED"); sendTelegramMessage(String(freshLogResult.data.staffTid), "❌ Your entry of ₹" + freshLogResult.data.amount + " was rejected by the owner." ); answerCallbackQuery(callbackId, "❌ Entry rejected"); editTelegramMessage(chatId, messageId, callbackQuery.message.text + "\n\n❌ *Rejected by owner*" ); } finally { releaseHotelLock(hotelId); } } // ── Delete request: APPROVE ── else if (data.startsWith("DEL_APPROVE_")) { var logId = data.substring(12); var fakeParams = { _telegramUser: { id: userId }, logId: logId, }; var approveResult = JSON.parse(handleApproveDelete(fakeParams).getContent()); if (!approveResult.success) { var approveMsg; if (approveResult.error === "NOT_PENDING_DELETE") { approveMsg = "⚠️ This entry is no longer pending delete (current status: " + approveResult.currentStatus + ")"; } else if (approveResult.error === "BUSY_TRY_AGAIN") { approveMsg = "⏳ Busy processing another request for this hotel — please tap again in a moment."; } else { approveMsg = "⚠️ " + approveResult.error; } answerCallbackQuery(callbackId, approveMsg); return; } answerCallbackQuery(callbackId, "✅ Delete approved"); editTelegramMessage(chatId, messageId, callbackQuery.message.text + "\n\n✅ *Deleted*" ); } // ── Delete request: REJECT ── else if (data.startsWith("DEL_REJECT_")) { var logId = data.substring(11); var fakeParams = { _telegramUser: { id: userId }, logId: logId, }; var rejectResult = JSON.parse(handleRejectDelete(fakeParams).getContent()); if (!rejectResult.success) { var rejectMsg; if (rejectResult.error === "NOT_PENDING_DELETE") { rejectMsg = "⚠️ This entry is no longer pending delete (current status: " + rejectResult.currentStatus + ")"; } else if (rejectResult.error === "BUSY_TRY_AGAIN") { rejectMsg = "⏳ Busy processing another request for this hotel — please tap again in a moment."; } else { rejectMsg = "⚠️ " + rejectResult.error; } answerCallbackQuery(callbackId, rejectMsg); return; } answerCallbackQuery(callbackId, "❌ Delete rejected"); editTelegramMessage(chatId, messageId, callbackQuery.message.text + "\n\n❌ *Delete rejected*" ); } else { answerCallbackQuery(callbackId, "Unknown action"); } } catch (err) { Logger.log("handleCallbackQuery error: " + err.message); answerCallbackQuery(callbackId, "⚠️ Error processing"); } } function sendWelcomeMessage(chatId, from) { var name = escapeMarkdown(from.first_name || "there"); sendTelegramMessage(chatId, "🙏 *Welcome to MyNirvaha, " + name + "!*\n\n" + "Your smart hotel expense tracker.\n\n" + "Tap the button below to open the app 👇" ); sendMiniAppButton(chatId, "📊 Open MyNirvaha", CONFIG.MINI_APP_URL); } function sendMiniAppButton(chatId, buttonText, url) { try { var apiUrl = TELEGRAM_API_BASE + CONFIG.BOT_TOKEN + "/sendMessage"; var payload = { chat_id: String(chatId), text: "👇", reply_markup: { inline_keyboard: [[ { text: buttonText, web_app: { url: url } } ]] } }; var options = { method: "post", contentType: "application/json", payload: JSON.stringify(payload), muteHttpExceptions: true, }; UrlFetchApp.fetch(apiUrl, options); } catch (err) { Logger.log("sendMiniAppButton error: " + err.message); } } function getBotUsername() { var cache = CacheService.getScriptCache(); var cached = cache.get("bot_username"); if (cached) return cached; try { var url = TELEGRAM_API_BASE + CONFIG.BOT_TOKEN + "/getMe"; var response = UrlFetchApp.fetch(url, { muteHttpExceptions: true }); var result = JSON.parse(response.getContentText()); if (result.ok && result.result.username) { cache.put("bot_username", result.result.username, 3600); return result.result.username; } } catch (err) { Logger.log("getBotUsername error: " + err.message); } return "MyNirvahaBot"; }
